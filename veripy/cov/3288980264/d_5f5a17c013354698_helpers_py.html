19def allowed_file(file_name, allowed_extensions): 

20 """A function to check the extension of the file upload.""" 

21 is_allowed = False 

22 if "." in file_name: 

23 extension = file_name.rsplit(".", 1)[1].lower() 

24 is_allowed = extension in allowed_extensions 

25 return is_allowed

28def upload_file(request, app_conf, file_name=None): 

29 # check if the POST request has the file part 

30 data = Data(None, []) 

31 if "upload" not in request.files: 

32 data.errors.append(gmsg(4101)) 

33 else: 

34 upload = request.files["upload"] 

35 # if a user does not select file, browser submits an empty part without filename 

36 if not upload.filename: 

37 data.errors.append(gmsg(4111)) 

38 elif upload: 

39 file_allowed = allowed_file(upload.filename, app_conf["ALLOWED_EXTENSIONS"]) 

40 if file_allowed: 

41 file_name = file_name if file_name else secure_filename(upload.filename) 

42 data.value = os.path.join(app_conf["UPLOAD_FOLDER"], file_name) 

43 upload.save(data.value) 

44 else: 

45 msg = ". Allowed extensions are %s" % ", ".join(app_conf["ALLOWED_EXTENSIONS"]) 

46 data.errors.append(gmsg(4091, msg)) 

47 return data

50def upload_gdocs_creds_file(request, app_conf): 

51 """Uploaded file will be stored in snippman/conf folder always as gdocs.json.""" 

52 data = upload_file(request, app_conf) 

53 if data.value: 

54 try: 

55 src = os.path.normpath(data.value) 

56 dst = os.path.normpath(app_conf["GDOCS_CREDS"]) 

57 creds_data = check_gdocs_creds_file(app_conf, src) 

58 if not creds_data.errors: 

59 if "win" in sys.platform and os.path.isfile(dst): # In Windows we should remove 

60 os.remove(dst) # the file if it exists 

61 os.rename(src, dst) # Move from upload folder to conf 

62 else: 

63 data.errors.extend(creds_data.errors) 

64 except OSError as err: 

65 data.errors.append(gmsg(1011, err)) # Could not complete upload due to OS error 

66 return data

83def load_to_db(data): 

84 """Rows are imported to SQLite DB one by one and skipped if very similar ones already exist.""" 

85 count = skipped = failed = 0 

86 flashes = [] 

87 for index, row in data: 

88 if len(row) < 8: 

89 flashes.append(fmsg(2152, row)) 

90 elif index > 0: # Skipping title row, so starting from 1 instead of 0 

91 row = [item.value if hasattr(item, "value") else item for item in row] 

92 msg, style, language = db_queries.create_language(row[0].lower()) 

93 if style == "success": 

94 flashes.append((msg, style)) 

95 values = [language.id, row[1], row[2], row[3]] 

96 msg, style, snippet = db_queries.create_snippet(values) 

97 values = {"created": row[4], "modified": row[5], "accessed": row[6], "visits": row[7]} 

98 db_queries.update_snippet_values(snippet.id, values) 

99 if style == "success": 

100 count += 1 

101 elif style == "warning": 

102 skipped += 1 

103 elif style == "danger": 

104 failed += 1 

105 if count > 0: 

106 style = 0 if (skipped + failed) == 0 else 3 

107 flashes.append(fmsg(2160 + style, count)) 

108 if skipped > 0: 

109 flashes.append(fmsg(2172, skipped)) 

110 if failed > 0: 

111 flashes.append(fmsg(2181, failed)) 

112 return flashes

146def update_settings_file(settings, conf_file_path, spreadsheet_id): 

147 """Settings are loaded from conf.py (which takes them from custom.json) at every launch. 

148 When settings are changed while app is running (through http://.../settings/update) - 

149 they are loaded into app.config dictionary. 

150 To "remember" them for next launches they are put into custom.json file. 

151 Hence, to avoid desynchronization, it is not recommended to edit custom.json manually. 

152 """ 

153 settings.update({"GDOCS_SPREADSHEET_ID": spreadsheet_id}) 

154 with open(conf_file_path, "w+") as _f: 

155 _f.write(json.dumps(settings)) 

156 return True
